 npm create vite@latest my-app -- --template react-ts
 cd my-app
 npm install
 nếu thiếu thư viện thì chạy npm install lucide-react
 copy 2 file  app. tsx và data.tsx vào

tạo mới một repo

 thêm vào file vite.config.ts đoạn code
export default defineConfig({
  plugins: [react()],
    base: "/thuy_linh/" // với thuy_linh là repo trên github
})

chạy lệnh npm i -D gh-pages

npm run build
